    <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
    <script>
        // Initialisation centrée sur Port-au-Prince
        var map = L.map('map', { zoomControl: false, attributionControl: false }).setView([18.5392, -72.3350], 12);

        // Simulation d'une zone électorale (Un carré orange pour Delmas)
        var delmasBounds = [[18.53, -72.35], [18.56, -72.31]];
        var zoneDelmas = L.rectangle(delmasBounds, {
            color: "#2C3E50", weight: 2, fillColor: "#E67E22", fillOpacity: 0.5
        }).addTo(map);

        // Ajout d'un texte "Delmas" au centre
        zoneDelmas.bindTooltip("Delmas", { permanent: true, direction: "center", className: "labels" }).openTooltip();

        // Gestion du clic pour FlutterFlow
        zoneDelmas.on('click', function() {
            var msg = "Delmas";
            if (window.FlutterFlowHoriz) { window.FlutterFlowHoriz.postMessage(msg); }
            else if (window.webkit && window.webkit.messageHandlers && window.webkit.messageHandlers.flutterFlow) {
                window.webkit.messageHandlers.flutterFlow.postMessage(msg);
            } else {
                alert("Clic sur : " + msg); // Pour tester sur ton navigateur d'ordinateur
            }
        });
    </script>
    <style>
        /* Style du texte Delmas (épuré, blanc, sans fond de boîte) */
        .labels { background: transparent; border: none; box-shadow: none; color: white; font-weight: bold; font-size: 14px; font-family: sans-serif; }
    </style>
